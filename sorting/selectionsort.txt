class Solution
{
	void select(int arr[], int i)
	{   
	     
	        for(int j=i+1;j<arr.length;j++)
	        {
	            if(arr[j]<arr[i])
	            {
	                int temp=arr[j];
	                arr[j]=arr[i];
	                arr[i]=temp;
	            }
	        }
        
	}
	
	
	   
	    
	
	
	void selectionSort(int arr[], int n)
	{ 
	    for(int i=0;i<n-1;i++)
	    {  select(arr,i);
	    }
	}
}
